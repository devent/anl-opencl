91 template<typename T>
92 cl_int copy(const cl::CommandQueue &queue, const cl::Buffer &buffer,
93  std::vector<T> &target) {
94  typedef typename std::vector<T>::value_type DataType;
95  const size_t byteLength = target.size() * sizeof(DataType);
96 
97  cl_int error;
98  DataType *pointer = static_cast<DataType*>(queue.enqueueMapBuffer(buffer,
99  CL_TRUE, CL_MAP_READ, 0, byteLength, 0, 0, &error));
100  // if exceptions enabled, enqueueMapBuffer will throw
101  if (error != CL_SUCCESS) {
102  return error;
103  }
104  std::memcpy(target.data(), pointer, byteLength);
105  cl::Event endEvent;
106  error = queue.enqueueUnmapMemObject(buffer, pointer, 0, &endEvent);
107  // if exceptions enabled, enqueueUnmapMemObject will throw
108  if (error != CL_SUCCESS) {
109  return error;
110  }
111  endEvent.wait();
112  return CL_SUCCESS;
113 }
