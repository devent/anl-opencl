104 template<typename T>
105 cl_int copy(const cl::CommandQueue &queue, const cl::Buffer &buffer,
106  std::vector<T> &target) {
107  typedef typename std::vector<T>::value_type DataType;
108  const size_t byteLength = target.size() * sizeof(DataType);
109 
110  cl_int error;
111  DataType *pointer = static_cast<DataType*>(queue.enqueueMapBuffer(buffer,
112  CL_TRUE, CL_MAP_READ, 0, byteLength, 0, 0, &error));
113  // if exceptions enabled, enqueueMapBuffer will throw
114  if (error != CL_SUCCESS) {
115  return error;
116  }
117  std::memcpy(target.data(), pointer, byteLength);
118  cl::Event endEvent;
119  error = queue.enqueueUnmapMemObject(buffer, pointer, 0, &endEvent);
120  // if exceptions enabled, enqueueUnmapMemObject will throw
121  if (error != CL_SUCCESS) {
122  return error;
123  }
124  endEvent.wait();
125  return CL_SUCCESS;
126 }
