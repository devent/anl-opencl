90 template<typename T>
91 cl_int copy(const cl::CommandQueue &queue, const cl::Buffer &buffer,
92  std::vector<T> &target) {
93  typedef typename std::vector<T>::value_type DataType;
94  const size_t byteLength = target.size() * sizeof(DataType);
95 
96  cl_int error;
97  DataType *pointer = static_cast<DataType*>(queue.enqueueMapBuffer(buffer,
98  CL_TRUE, CL_MAP_READ, 0, byteLength, 0, 0, &error));
99  // if exceptions enabled, enqueueMapBuffer will throw
100  if (error != CL_SUCCESS) {
101  return error;
102  }
103  std::memcpy(target.data(), pointer, byteLength);
104  cl::Event endEvent;
105  error = queue.enqueueUnmapMemObject(buffer, pointer, 0, &endEvent);
106  // if exceptions enabled, enqueueUnmapMemObject will throw
107  if (error != CL_SUCCESS) {
108  return error;
109  }
110  endEvent.wait();
111  return CL_SUCCESS;
112 }
